🧠 3. BACKEND (index.js)

Paste this:

import express from "express";
import cors from "cors";

const app = express();
app.use(cors());
app.use(express.json());
app.use(express.static("public"));

app.post("/generate", async (req, res) => {
  const { signal } = req.body;

  // simple intent simulation (replace with OpenAI later)
  let response = "That feels right.";

  if (signal > 0.7) response = "Let’s move forward.";
  else if (signal < 0.4) response = "Something is off.";

  res.json({ text: response, confidence: signal });
});

app.listen(3000, () => console.log("Server running"));

⚡ 6. LOGIC (public/app.js)

const orb = document.getElementById("orb");
const text = document.getElementById("text");

let holdStart;

orb.addEventListener("mousedown", () => {
  holdStart = Date.now();
  orb.classList.add("active");
});

orb.addEventListener("mouseup", async () => {
  orb.classList.remove("active");

  const duration = (Date.now() - holdStart) / 1000;
  const signal = Math.min(duration, 1); // normalize

  text.innerText = "Thinking...";

  const res = await fetch("/generate", {
    method: "POST",
    headers: { "Content-Type": "application/json" },
    body: JSON.stringify({ signal })
  });

  const data = await res.json();

  text.innerText = data.text;

  // voice output
  const speech = new SpeechSynthesisUtterance(data.text);
  speechSynthesis.speak(speech);

  // confidence UI
  if (data.confidence < 0.5) {
    text.style.opacity = 0.5;
  } else {
    text.style.opacity = 1;
  }
});
