<script>// Playables SDK v1.0.0
// Game lifecycle bridge: rAF-based game-ready detection + event communication
(function() {
  'use strict';

  // Idempotency: skip if already initialized (e.g., server-side injection
  // followed by client-side inject-javascript via the Bloks webview component).
  if (window.playablesSDK) return;

  var HANDLER_NAME = 'playablesGameEventHandler';
  var ANDROID_BRIDGE_NAME = '_MetaPlayablesBridge';
  var RAF_FRAME_THRESHOLD = 3;

  var gameReadySent = false;
  var firstInteractionSent = false;
  var errorSent = false;
  var frameCount = 0;
  var originalRAF = window.requestAnimationFrame;

  // --- Transport Layer ---

  function hasIOSBridge() {
    return !!(window.webkit &&
              window.webkit.messageHandlers &&
              window.webkit.messageHandlers[HANDLER_NAME]);
  }

  function hasAndroidBridge() {
    return !!(window[ANDROID_BRIDGE_NAME] &&
              typeof window[ANDROID_BRIDGE_NAME].postEvent === 'function');
  }

  function isInIframe() {
    return !!(window.parent && window.parent !== window);
  }

  function sendEvent(eventName, payload) {
    var message = {
      type: eventName,
      payload: payload || {},
      timestamp: Date.now()
    };

    if (hasIOSBridge()) {
      try {
        window.webkit.messageHandlers[HANDLER_NAME].postMessage(message);
      } catch (e) { /* ignore */ }
      return;
    }

    if (hasAndroidBridge()) {
    try {
      var p = payload || {};
      p.__secureToken = window.__fbAndroidBridgeAuthToken || '';
      p.timestamp = message.timestamp;
      window[ANDROID_BRIDGE_NAME].postEvent(
        eventName,
        JSON.stringify(p)
      );
    } catch (e) { /* ignore */ }
    return;
  }

    if (isInIframe()) {
      try {
        window.parent.postMessage(message, '*');
      } catch (e) { /* ignore */ }
      return;
    }
  }

  // --- rAF Game-Ready Detection ---

  function onFrame() {
    if (gameReadySent) return;

    frameCount++;
    if (frameCount >= RAF_FRAME_THRESHOLD) {
      gameReadySent = true;
      sendEvent('game_ready', {
        frame_count: frameCount,
        detected_at: Date.now()
      });
      return;
    }

    originalRAF.call(window, onFrame);
  }

  if (originalRAF) {
    window.requestAnimationFrame = function(callback) {
      if (!gameReadySent) {
        return originalRAF.call(window, function(timestamp) {
          frameCount++;
          if (frameCount >= RAF_FRAME_THRESHOLD && !gameReadySent) {
            gameReadySent = true;
            sendEvent('game_ready', {
              frame_count: frameCount,
              detected_at: Date.now()
            });
          }
          callback(timestamp);
        });
      }
      return originalRAF.call(window, callback);
    };
  }

  // --- First User Interaction Detection ---

  function setupFirstInteractionDetection() {
    var events = ['touchstart', 'mousedown', 'keydown'];

    function onFirstInteraction() {
      if (firstInteractionSent) return;
      firstInteractionSent = true;
      sendEvent('user_interaction_start', null);

      for (var i = 0; i < events.length; i++) {
        document.removeEventListener(events[i], onFirstInteraction, true);
      }
    }

    for (var i = 0; i < events.length; i++) {
      document.addEventListener(events[i], onFirstInteraction, true);
    }
  }

  if (document.readyState === 'loading') {
    document.addEventListener('DOMContentLoaded', setupFirstInteractionDetection);
  } else {
    setupFirstInteractionDetection();
  }

  // --- Auto Error Capture ---

  window.addEventListener('error', function(event) {
    if (errorSent) return;
    errorSent = true;
    sendEvent('error', {
      message: event.message || 'Unknown error',
      source: event.filename || '',
      lineno: event.lineno || 0,
      colno: event.colno || 0,
      auto_captured: true
    });
  });

  window.addEventListener('unhandledrejection', function(event) {
    if (errorSent) return;
    errorSent = true;
    var reason = event.reason;
    sendEvent('error', {
      message: (reason instanceof Error) ? reason.message : String(reason),
      type: 'unhandled_promise_rejection',
      auto_captured: true
    });
  });

  // --- Public API ---

  window.playablesSDK = {
    complete: function(score) {
      sendEvent('game_ended', {
        score: score,
        completed: true
      });
    },

    error: function(message) {
      if (errorSent) return;
      errorSent = true;
      sendEvent('error', {
        message: message || 'Unknown error',
        auto_captured: false
      });
    },

    sendEvent: function(eventName, payload) {
      if (!eventName || typeof eventName !== 'string') return;
      sendEvent(eventName, payload);
    }
  };

  // Kick off rAF detection in case no game code calls rAF immediately
  if (originalRAF) {
    originalRAF.call(window, onFrame);
  }
})();</script>

function americanToDecimal(odds) {
  const o = parseInt(odds);
  return o > 0 ? (o / 100) + 1 : (100 / Math.abs(o)) + 1;
}
function decimalToAmerican(dec) {
  if (dec >= 2) return '+' + Math.round((dec - 1) * 100);
  return '-' + Math.round(100 / (dec - 1));
}
function toast(msg, color = '#ff8c42') {
  const wrap = document.getElementById('toasts');
  const el = document.createElement('div');
  el.className = 'panel rounded-lg p-3 text-[12px] leading-snug shadow-2xl translate-x-2 opacity-0 transition-all duration-300';
  el.style.borderLeft = `3px solid ${color}`;
  el.innerHTML = `<div class="flex items-start gap-2"><div class="w-1.5 h-1.5 rounded-full mt-1.5" style="background:${color}"></div><div><div class="font-medium mb-0.5">NEXUS ORACLE</div><div class="text-[#9aa4b2]">${msg}</div></div></div>`;
  wrap.appendChild(el);
  requestAnimationFrame(() => { el.classList.remove('translate-x-2', 'opacity-0'); });
  setTimeout(() => { el.classList.add('translate-x-2','opacity-0'); setTimeout(()=>el.remove(),300); }, 2800);
}

function updateParlay() {
  parlayCount.textContent = `${parlay.length} LEG${parlay.length!==1?'S':''}`;
  if (parlay.length === 0) {
    parlayEmpty.classList.remove('hidden');
    parlayList.classList.add('hidden');
    parlaySummary.classList.add('hidden');
    return;
  }
  parlayEmpty.classList.add('hidden');
  parlayList.classList.remove('hidden');
  parlaySummary.classList.remove('hidden');

  parlayList.innerHTML = parlay.map((p,i)=>`
    <div class="group flex items-center justify-between p-2.5 rounded-lg bg-[#0c0f14] border border-[#1f242f] hover:border-[#2a3140]">
      <div class="min-w-0">
        <div class="text-[12px] font-medium truncate">${p.prop}</div>
        <div class="flex items-center gap-2 mt-0.5">
          <span class="text-[10px] mono text-[#7a8699]">${p.game}</span>
          <span class="text-[9px] px-1 py-0.5 rounded ${p.path==='DRAGON'?'bg-[#ff8c42]/15 text-[#ff8c42]':p.path==='MIRROR'?'bg-[#5aa9e6]/15 text-[#5aa9e6]':'bg-[#ff5c8a]/15 text-[#ff5c8a]'}">${p.path}</span>
        </div>
      </div>
      <div class="flex items-center gap-2.5">
        <span class="mono text-[12px]">${p.odds>0?'+'+p.odds:p.odds}</span>
        <button data-idx="${i}" class="remove w-5 h-5 grid place-items-center rounded hover:bg-[#1f242f] text-[#7a8699] hover:text-white transition">×</button>
      </div>
    </div>
  `).join('');

  parlayList.querySelectorAll('.remove').forEach(btn=>{
    btn.onclick = () => {
      const idx = +btn.dataset.idx;
      const removed = parlay.splice(idx,1)[0];
      cards.forEach(c=>{ if(c.dataset.prop===removed.prop) c.classList.remove('selected'); });
      updateParlay();
      toast(`Removed ${removed.prop}`, '#7a8699');
    };
  });

  let dec = 1, edgeSum = 0;
  parlay.forEach(p=>{ dec *= americanToDecimal(p.odds); edgeSum += parseFloat(p.edge); });
  const combined = decimalToAmerican(dec);
  combinedOddsEl.textContent = combined;
  totalEdgeEl.textContent = `+${(edgeSum).toFixed(1)}%`;
  payoutEl.textContent = `$${(100*dec).toFixed(2)}`;

  // risk
  const games = parlay.map(p=>p.game);
  const dupes = games.length - new Set(games).size;
  let risk = 'LOW', pct = 20, color = '#3ddc97';
  if (parlay.length >= 4 || dupes >= 2) { risk = 'HIGH'; pct = 85; color = '#ff5c5c'; }
  else if (parlay.length >= 3 || dupes === 1) { risk = 'MED'; pct = 55; color = '#ffbf5c'; }
  riskLabel.textContent = risk;
  riskBar.style.width = pct + '%';
  riskBar.style.background = color;
}

cards.forEach(card=>{
  card.addEventListener('click', ()=>{
    const data = {
      player: card.dataset.player,
      prop: card.dataset.prop,
      odds: parseInt(card.dataset.odds),
      game: card.dataset.game,
      path: card.dataset.path,
      edge: card.dataset.edge
    };
    const exists = parlay.findIndex(p=>p.prop===data.prop);
    if (exists >=0) {
      parlay.splice(exists,1);
      card.classList.remove('selected');
      toast(`Removed ${data.prop} from assembler`, '#7a8699');

      parlay.push(data);

      toast(`Launching ${data.player} path • ${data.path}`, data.path==='DRAGON'?'#ff8c42':data.path==='MIRROR'?'#5aa9e6':'#ff5c8a');
      // pulse lattice
      if (window.pulseNode) window.pulseNode(data.player);
    }
    updateParlay();
  });
});

function resize(){
  const w = canvas.clientWidth, h = canvas.clientHeight;
  if(!w||!h) return;
  renderer.setSize(w,h,false);
  camera.aspect = w/h;
  camera.updateProjectionMatrix();
}
new ResizeObserver(resize).observe(canvas);
resize();

window.pulseNode = (name)=>{
  const key = Object.keys(nodeMap).find(k=>name.includes(players.find(p=>p.id===k).name) || name.includes('Towns')&&k==='KAT' || name.includes('Alexander')&&k==='NAW');
  const m = nodeMap[key||'KAT'];
  if(!m) return;
  const start = Date.now();
  const base = m.scale.x;
  function pulse(){ const t=(Date.now()-start)/600; if(t>1){m.scale.setScalar(base);return;} m.scale.setScalar(base+Math.sin(t*Math.PI)*0.8); requestAnimationFrame(pulse);} pulse();
};

let t=0;

  t+=0.002;
  group.rotation.y = Math.sin(t*0.3)*0.15 + t*0.05;
  group.rotation.x = Math.sin(t*0.2)*0.08;
  renderer.render(scene,camera);
  requestAnimationFrame(animate);
}
